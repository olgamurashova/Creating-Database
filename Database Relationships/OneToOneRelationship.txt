// Enforcing a strictly one-to-one relationship using UNIQUE keyword
1. CREATE TABLE driver (
    license_id char(20) PRIMARY KEY,
    name varchar(20),
    address varchar(100),
    date_of_birth date
);      
 
CREATE TABLE license (
    id integer PRIMARY KEY,
    state_issued varchar(20),
    date_issued date,
    date_expired  date,
    license_id char(20) REFERENCES driver(license_id) UNIQUE
);

2. CREATE TABLE book_details (
  id integer PRIMARY KEY,
  book_isbn varchar(50) REFERENCES book(isbn) UNIQUE,
  rating decimal,
  language varchar(10),
  keywords text[],
  date_published date
);

SELECT 
constraint_name, table_name, column_name
FROM
information_schema.key_column_usage
WHERE
table_name = 'book_details';

SELECT 
book.title, book.price, book_details.language, book_details.rating
FROM book, book_details
WHERE book.isbn = book_details.book_isbn;

OR

SELECT
book.title, book.price, book_details.language, book_details.rating
FROM book
JOIN book_details
ON book.isbn = book_details.book_isbn;
